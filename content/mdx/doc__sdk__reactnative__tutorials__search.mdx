This guide explains how to manage the Search for Rainbow contacts,conversation or Bubbles ...etc. So basically we build two component foe the search: `Search Bar Input` and the `Search Result View`.

#### Search Bar input.

---

This will show you a search text input.

#

Name

Notes

Default

Description

1

onSearchUpdated

(value: boolean, searchQuery: string): void

none

handle when the search text is updated

2

onCancelSearch

() => void

none

handle when canceling the search

3

registerBackButtonHandler

(handler: [IBackButtonHandler](/docs/sdk/reactnative/api/IActions#IBackButtonHandler)) => void

none

used for registering the back button in the search bar

We usually put the Search Bar input in the header \[as its show in the `Search Input Example`\]

#### Search Result View Example.

```tsx
import { View } from 'react-native';
import React, { useState } from 'react';
import { OpenCallDialogComponent, PbxActions } from './CallComponent';
import {
  createCallPbxContactAction,
  IAction,
  IActionsProvider,
  IContact,
  Search,
} from 'react-native-rainbow-module';

export const SearchComponent: React.FunctionComponent = ({navigation}) => {
  const [showDialog, setShowDialog] = useState<boolean>(false);
  const [contactToCall, setContactToCall] = useState<IContact>();

  const openDialog = () => {
    setShowDialog(!showDialog);
  };
  const actions: IAction[] = [
    createCallPbxContactAction((contact) => {
      setContactToCall(contact);
      openDialog();
    }),
  ];

  const contactActionsProvider: IActionsProvider = new (class {
    public getActions = (contact: IContact) => {
      return actions;
    };
  })();
  const handleClickItem = (contact: IContact) => {
        navigation.navigate('ContactInformation', {contact})

  };
  return (
    <View>
      <Search
        searchPeople={true}
        contactActionsProvider={contactActionsProvider}
        onClickItem={handleClickItem}
      />
      {contactToCall && (
        <OpenCallDialogComponent
          contact={contactToCall}
          showDialog={showDialog}
          openDialog={openDialog}
          callDialogWebRtcActions={[]}
          callDialogPbxActions={PbxActions}
        />
      )}
    </View>
  );
};
```
